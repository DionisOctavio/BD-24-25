**** SELECT ARIGMETICOS ****

SELECT ABS(-10) FROM DUAL;

SELECT CEIL(12.08) FROM DUAL;

SELECT FLOOR(12.08) FROM DUAL;

SELECT ROUND(12.08) FROM DUAL; // REDONDEO NATURAL

SELECT POWER(12,2) FROM DUAL; //DOS PARAMETROS SEPARADOS POR COMAS

SELECT MOD(12,2) FROM DUAL; // COJO EL PRIMERO LO DIVIDO POR EL SEGUNDO Y ME DA EL RESTO

SELECT SIGN(12) FROM DUAL; // EL SIGNO DE UN NUMERO

SELECT SQRT(12) FROM DUAL; // RAIZ CUADRADA DE UN NUMERO

SELECT TRUNC(1299.256,-2) FROM DUAL; // ELIMINA LOS DECIMALES OSEA RECORTA
 
**** CONBINACION ****

SELECT ROUND (SQRT(12), 2) FROM DUAL; // RAIZ CUADRADA REDONDEANDO POR UN NUMERO

SELECT ROUND (SQRT(1254), -2) FROM DUAL; // TAMBIEN SE PUEDE REDONDEAR A NEGATIVO 

***** SELECT *****

// APELLIDO Y SALARIO DEPT 10

SELECT APELLIDO, SALARIO
FROM EMPLE
WHERE DEPT_NO = 10;



// APELLIDO, SALARIO Y COMISION DPT 10

SELECT APELLIDO, SALARIO, COMISION
FROM EMPLE
WHERE DEPT_NO = 10;


// SALSRIO APELLIDO COMISION Y LA SUMA TOTAL DE ESTOS

SELECT APELLIDO, SALARIO, COMISION, SALARIO+COMISION AS SUELDO_TOTAL
FROM EMPLE
WHERE DEPT_NO = 30;


// COMO HAY COMISIONES NULL LA SUMA FALLA ASI QUE VASOM A SOLUCIONARLO CAMBIANDO NULL POR 0 (NVL ES UNA FUNCION QUE SI ES NUL CAMBIA EL VALO POR LO QUE TU QUIERAS PAG 76)

SELECT APELLIDO, SALARIO, COMISION, SALARIO+NVL (COMISION,0) AS SUELDO_TOTAL
FROM EMPLE
WHERE DEPT_NO = 30;
