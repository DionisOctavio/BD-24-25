SELECT EMP_NO, APELLIDO, SALARIO
FROM EMPLE
WHERE SALARIO IN (SELECT MAX(SALARIO)
FROM EMPLE WHERE EMPLE.DEPT_NO = 20) 
AND DEPT_NO = 20;


SELECT EMP_NO, APELLIDO, DEPART.DNOMBRE, SALARIO
FROM EMPLE
JOIN DEPART ON EMPLE.DEPT_NO = DEPART.DEPT_NO
WHERE SALARIO IN (SELECT MAX(SALARIO)
FROM EMPLE WHERE EMPLE.DEPT_NO = DEPART.DEPT_NO) 
AND EMPLE.DEPT_NO = DEPART.DEPT_NO;


SELECT EMP_NO, APELLIDO, DEPART.DNOMBRE, SALARIO
FROM EMPLE
WHERE SALARIO IN (SELECT MAX(SALARIO)
FROM EMPLE WHERE EMPLE.DEPT_NO = DEPART.DEPT_NO) 
AND EMPLE.DEPT_NO = DEPART.DEPT_NO;


SELECT EMP_NO, APELLIDO, DEPT_NO, SALARIO
FROM EMPLE k
WHERE SALARIO = (SELECT MAX(SALARIO) FROM EMPLE WHERE EMPLE.DEPT_NO = K.DEPT_NO)
ORDER BY SALARIO DESC;
